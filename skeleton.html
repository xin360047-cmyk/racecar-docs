代码骨架：跟着这个顺序写

不是让你一次看完——按顺序从下往上一层一层搭，每一层写完都能单独测试
你应该按这个顺序写代码
1. bsp/
→
2. module/
→
3. algo/
→
4. config/
→
5. app/
→
6. main.c
为什么是这个顺序？
  • bsp/ 先写——你连 PWM 都输出不了、ADC 都读不到值，后面写什么都没用
  • module/ 第二——把 bsp 包一层，变成"line_pos""distance""heading"这些有意义的变量
  • algo/ 第三——纯数学的 PID，不依赖任何硬件，可以在电脑上单独调通
  • config/ 第四——把所有可调的数字集中到一个头文件
  • app/ 第五——状态机和主控循环，把前四层拼起来
  • main.c 最后——只做初始化 + 启动循环，5 行有效代码
第一步：bsp/ — 让每个外设"能跑"
这一层不写任何业务逻辑。目标：串口能打印、ADC 能读到电压、定时器能输出 PWM、编码器能计数、I2C 能读到设备地址。每个文件写完立刻单独测。
bsp/ 不关心"用这个外设做什么"，只关心"这个外设本身对不对"。
/* ====== bsp_adc.h — 灰度传感器 8 路 ADC ====== */ #ifndef BSP_ADC_H #define BSP_ADC_H #include <ti/driverlib/driverlib.h> // MSPM0 DriverLib /* 8 路灰度对应的 ADC 通道 */ #define ADC_CH_GRAY_0 DL_ADC12_MEM_IDX_0 #define ADC_CH_GRAY_1 DL_ADC12_MEM_IDX_1 #define ADC_CH_GRAY_2 DL_ADC12_MEM_IDX_2 #define ADC_CH_GRAY_3 DL_ADC12_MEM_IDX_3 #define ADC_CH_GRAY_4 DL_ADC12_MEM_IDX_4 #define ADC_CH_GRAY_5 DL_ADC12_MEM_IDX_5 #define ADC_CH_GRAY_6 DL_ADC12_MEM_IDX_6 #define ADC_CH_GRAY_7 DL_ADC12_MEM_IDX_7 void bsp_adc_init(void); // 初始化 ADC + 8 个通道 uint16_t bsp_adc_read_ch(uint8_t channel); // 读单通道原始值 #endif
/* ====== bsp_adc.c ====== */ void bsp_adc_init(void) { // 1. 开 ADC 时钟 (DL_ADC12) // 2. 配 8 个模拟输入引脚 // 3. 配 ADC 采样时间 + 分辨率 (12bit) // 4. 使能 ADC // ⚠ 用 TI DriverLib，不手写寄存器。CCS SysConfig 可以自动生成这部分 } uint16_t bsp_adc_read_ch(uint8_t ch) { // DL_ADC12_startConversion() → 等完成 → DL_ADC12_getMemResult(ch) return DL_ADC12_getMemResult(ch); // 返回 0~4095 } /* ✅ 测法：串口打印 8 路值，手拿白纸和黑胶带分别在传感器下划过，看数值变不变 */
/* ====== bsp_tim.h — 定时器 ====== */ void bsp_pwm_init(void); // PWM 输出 (TIMG) → 520 电机 void bsp_encoder_init(void); // 编码器模式 (TIMG 正交解码) void bsp_pwm_set_l(uint16_t duty); // 0~1000 void bsp_pwm_set_r(uint16_t duty); int32_t bsp_encoder_get_l(void); // 读编码器计数值 int32_t bsp_encoder_get_r(void); /* ✅ 测法：PWM → 示波器看波形。编码器 → 手转轮子串口打印计数 */
/* ====== bsp_i2c.h — I2C (OLED + MPU6050) ====== */ void bsp_i2c_init(void); uint8_t bsp_i2c_write(uint8_t addr, uint8_t reg, uint8_t data); uint8_t bsp_i2c_read(uint8_t addr, uint8_t reg, uint8_t *buf, uint8_t len); /* ✅ 测法：I2C 扫设备地址 → 串口打印 0x3C(OLED) 0x68(MPU6050) 看到就算通 */
/* ====== bsp_uart.h — 串口调试 ====== */ void bsp_uart_init(void); void bsp_uart_send(uint8_t *data, uint16_t len); // 重定向 printf 到串口，调试全靠它 // ✅ 测法：printf("Hello\r\n"); 串口助手收到就是通
/* ====== bsp_gpio.h — 开关量 ====== */ #define BUZZER_PIN /* 查 pin_config.h */ #define LED_PIN #define BTN_START_PIN #define BUZZER_ON() DL_GPIO_setPins(BUZZER_PORT, BUZZER_PIN) #define BUZZER_OFF() DL_GPIO_clearPins(BUZZER_PORT, BUZZER_PIN) #define BTN_START() (!DL_GPIO_readPins(BTN_PORT, BTN_PIN)) /* ✅ 测法：按键 → 蜂鸣器响 → 松开停。5 秒测完 */
第二步：module/ — 把传感器读数变成有意义的变量
这一层调用 bsp/，但不再暴露"ADC 通道几""编码器计数"这些底层东西。对外只输出三个变量：line_pos、distance、heading，外加两个写入接口：pwm_l、pwm_r，和一个提示接口：beep。
/* ====== mod_grayscale.c — 8 路 ADC → line_pos (-100~+100) ====== */ #include "bsp/bsp_adc.h" static uint16_t black[8], white[8]; // 校准值 void mod_grayscale_calibrate(void) { // 车放黑线上 → 读 8 路 → 存到 black[] // 车放白布上 → 读 8 路 → 存到 white[] // 阈值 = (black[i] + white[i]) / 2 } int16_t mod_grayscale_read(void) { // 1. 读 8 路 ADC 值 // 2. 每路跟自己的阈值比：低于阈值→在黑线上(1)，高于→白地(0) // 3. 加权求和：最左 -4权重，最右 +4权重，中间 0 权重 // 8 路全黑 → line_pos ≈ 0（黑线在正中间） // 右边几路黑 → line_pos > 0（偏右，需左转） // 左边几路黑 → line_pos < 0（偏左，需右转） return line_pos; // -100 到 +100 } /* ✅ 测法：车静止，手动左右移动黑胶带，串口打印 line_pos 看变化 */
/* ====== mod_encoder.c — 编码器计数 → speed + distance ====== */ #include "config/car_config.h" #include "bsp/bsp_tim.h" static int32_t last_cnt_l, last_cnt_r; static float total_distance; void mod_encoder_reset(void) { total_distance = 0; } float mod_encoder_get_distance(void) { // 每次定时器中断 (比如 10ms)： // delta_l = enc_l - last_cnt_l; 同步清零 // delta_r = enc_r - last_cnt_r; // avg_delta = (delta_l + delta_r) / 2 // dist_step = avg_delta / ENCODER_PPR * π * WHEEL_DIAMETER // total_distance += dist_step return total_distance; } float mod_encoder_get_speed_l(void) { /* delta_l → mm/s */ } float mod_encoder_get_speed_r(void) { /* delta_r → mm/s */ } /* ✅ 测法：手推车走 1m，串口打印 distance，看是不是 ≈1000mm */
/* ====== mod_motor.c — PWM 占空比 → 四路编码器驱动 → 520 电机 ====== */ #include "bsp/bsp_tim.h" void mod_motor_set(uint16_t pwm_l, uint16_t pwm_r) { // pwm_l → 左轮 (正转 = 车前进) // pwm_r → 右轮 (正转 = 车前进) // ⚠ 四路编码器驱动模块一个 PWM 控制一个电机方向+速度 // 如果接了 DIR 脚：pwm > 0 → DIR=高(前进)；pwm=绝对值 bsp_pwm_set_l(pwm_l); bsp_pwm_set_r(pwm_r); } void mod_motor_stop(void) { bsp_pwm_set_l(0); bsp_pwm_set_r(0); } /* ✅ 测法：mod_motor_set(300,300) → 两轮同速正转，车直走 */
/* ====== mod_imu.c — MPU6050/JY901 → heading ====== */ void mod_imu_calibrate(void); // 静止采零漂，存 gyro_bias float mod_imu_get_heading(void); // 累计陀螺仪 Z 轴积分 → 航向角 0~360° /* ✅ 测法：串口打印 heading，转车 90° 看数字是否变 ≈90 */
/* ====== mod_buzzer.c — 声光提示 ====== */ void mod_buzzer_beep(void) { BUZZER_ON(); DL_Common_delayCycles(8000000); BUZZER_OFF(); // 100ms } /* ✅ 测法：调用一次响一声 */
第三步：algo/ — PID 纯数学，可以在电脑上测
这一层 不 #include 任何硬件头文件。输入是数字，输出也是数字。你甚至可以在电脑上写个 Python 版先验算。
/* ====== pid.c — 两个 PID ====== */ #include "pid.h" #include "config/param.h" // 只引参数，不引硬件 /* 方向环：位置式 PID。输入 error（偏离多少），输出转向量 */ int16_t pid_direction(float error, float kp, float kd) { static float last = 0; float out = kp * error + kd * (error - last); last = error; return (int16_t)out; } /* 速度环：增量式 PID。输入 target 和 actual，输出 PWM */ uint16_t pid_speed(float target, float actual, uint8_t side) { static float e1[2]={0,0}, e2[2]={0,0}, u[2]={0,0}; float e = target - actual; float inc = SPD_KP*(e-e1[side]) + SPD_KI*e + SPD_KD*(e-2*e1[side]+e2[side]); u[side] += inc; if (u[side] > 1000) u[side] = 1000; // 限幅 if (u[side] < 0) u[side] = 0; e2[side] = e1[side]; e1[side] = e; return (uint16_t)u[side]; } /* ✅ 测法（脱离硬件）：写个 main() 给一串假 line_pos → 打印 turn_amount * 看：正误差得负输出、负误差得正输出 → 逻辑对 */
第四步：config/ — 把所有调得到的数字集中到三个头文件
这一层没有任何 .c 文件，全是 #define。目标：换一个场地、换一套传感器，只改这三个文件。
/* ====== config/pin_config.h — 引脚映射 ====== */ // ADC 引脚（8 路灰度） // PWM 引脚（左右电机） // I2C 引脚（OLED + MPU6050） // UART 引脚（JY901 / 调试串口） // GPIO 引脚（蜂鸣器 / LED / 按键） // 全部宏定义，查天猛星原理图后填 /* ====== config/car_config.h — 小车物理参数 ====== */ #define WHEEL_DIAMETER 65 // mm，卡尺量 #define ENCODER_PPR 330 // 520 电机编码器每圈脉冲，串口验证 #define TRACK_WIDTH 130 // 左右轮中心距 mm /* ====== config/param.h — 运行时参数（赛场只改这个）===== */ // --- 速度 --- #define BASE_SPEED 600 // 直道/对角线 PWM #define ARC_SPEED 350 // 弧线段 PWM（直道 60%） #define TURN_LIMIT 300 // 最大转向量 // --- 方向环 --- #define DIR_KP 2.5 #define DIR_KP_ARC 5.0 // 弧线大 P（直道 2 倍） #define DIR_KD 0.8 // --- 速度环 --- #define SPD_KP 3.0 #define SPD_KI 0.15 #define SPD_KD 0.5 // --- 场地 --- #define SEG_DIAG 360 // A→C 对角线 mm #define SEG_ARC 377 // 弧线半周长 mm（π×40cm≈1256mm 的半弧需要赛场量） #define VERTEX_TOL 20 // 顶点容差 mm
第五步：app/ — 把前四层拼成赛道行为
这一层不再碰硬件。只做两件事：① 决定"现在处于哪一段"（状态机），② 在每个状态下怎么控制（task_control）。
app/ 离硬件最远，但离"车的行为"最近。赛道逻辑全在这。
/* ====== app/state_machine.h ====== */ typedef enum { STATE_DIAG_AC, // A→C 对角线 · 白地 · IMU 航向锁定 STATE_ARC_CB, // C→B 右弧线 · 巡黑线 · 降速+大P STATE_DIAG_BD, // B→D 对角线 · 白地 · IMU 航向锁定 STATE_ARC_DA, // D→A 左弧线 · 巡黑线 · 降速+大P STATE_STOP // 4圈完成 · 停车 } State; extern State g_state; extern uint8_t g_lap; void state_init(void); // state=S0, lap=0 void state_update(float distance); // 传入里程 → 判定是否切状态 /* ====== app/task_control.h ====== */ void task_control(void); // 主控循环：读→算→出→判。每 10ms 调一次
/* ====== app/state_machine.c ====== */ #include "state_machine.h" #include "config/param.h" #include "module/mod_encoder.h" #include "module/mod_buzzer.h" State g_state = STATE_DIAG_AC; uint8_t g_lap = 0; void state_init(void) { g_state = STATE_DIAG_AC; g_lap = 0; mod_encoder_reset(); } void state_update(float distance) { switch (g_state) { case STATE_DIAG_AC: if (distance >= SEG_DIAG) { g_state = STATE_ARC_CB; mod_encoder_reset(); mod_buzzer_beep(); } break; case STATE_ARC_CB: if (distance >= SEG_ARC) { g_state = STATE_DIAG_BD; mod_encoder_reset(); mod_buzzer_beep(); } break; case STATE_DIAG_BD: if (distance >= SEG_DIAG) { g_state = STATE_ARC_DA; mod_encoder_reset(); mod_buzzer_beep(); } break; case STATE_ARC_DA: if (distance >= SEG_ARC) { mod_buzzer_beep(); g_lap++; if (g_lap >= 4) g_state = STATE_STOP; else { g_state = STATE_DIAG_AC; mod_encoder_reset(); } } break; case STATE_STOP: mod_motor_stop(); break; } }
/* ====== app/task_control.c — 主控循环，每 10ms 跑一次 ====== */ #include "task_control.h" #include "state_machine.h" #include "config/param.h" #include "module/mod_grayscale.h" #include "module/mod_encoder.h" #include "module/mod_motor.h" #include "module/mod_imu.h" #include "algo/pid.h" void task_control(void) { // ===== 1. 读：从 module 拿三个核心变量 ===== int16_t line_pos = mod_grayscale_read(); float distance = mod_encoder_get_distance(); float heading = mod_imu_get_heading(); float spd_l = mod_encoder_get_speed_l(); float spd_r = mod_encoder_get_speed_r(); if (g_state == STATE_STOP) return; // ===== 2. 算：根据当前状态选策略 → PID ===== uint16_t base; int16_t turn; if (g_state == STATE_ARC_CB || g_state == STATE_ARC_DA) { // 弧线段：有黑线，用灰度巡线。降速 + 大 P base = ARC_SPEED; turn = pid_direction(line_pos, DIR_KP_ARC, DIR_KD); } else { // 对角线段：白地，没有黑线。锁定陀螺仪航向 base = BASE_SPEED; static float target_h; if (distance < 5) target_h = heading; // 刚进入状态时锁当前航向 float h_err = target_h - heading; turn = pid_direction(h_err, DIR_KP, DIR_KD); } // ===== 3. 出：速度环 → PWM ===== int16_t tL = base + turn, tR = base - turn; if (tL > 1000) tL = 1000; if (tL < 0) tL = 0; if (tR > 1000) tR = 1000; if (tR < 0) tR = 0; uint16_t pwmL = pid_speed(tL, spd_l, 0); uint16_t pwmR = pid_speed(tR, spd_r, 1); mod_motor_set(pwmL, pwmR); // ===== 4. 判：里程到了就切状态 ===== state_update(distance); }
第六步：main.c — 只做初始化 + 启动循环
main.c 是所有分层的结果——它自己没有任何逻辑，只负责把各层初始化、然后死循环跑 task_control()。就 5 行有效代码。
/* ====== main.c — 入口 ====== */ #include "bsp/bsp_clock.h" #include "bsp/bsp_tim.h" #include "bsp/bsp_adc.h" #include "bsp/bsp_i2c.h" #include "bsp/bsp_uart.h" #include "bsp/bsp_gpio.h" #include "module/mod_grayscale.h" #include "module/mod_encoder.h" #include "module/mod_motor.h" #include "module/mod_imu.h" #include "module/mod_oled.h" #include "app/state_machine.h" #include "app/task_control.h" void main(void) { /* --- 初始化：从下往上 Init --- */ bsp_clock_init(); // SYSCLK = 80MHz bsp_tim_init(); // PWM + 编码器模式 bsp_adc_init(); // 灰度 8 路 ADC bsp_i2c_init(); // OLED + MPU6050 bsp_uart_init(); // 调试串口 + printf bsp_gpio_init(); // 蜂鸣器 LED 按键 mod_grayscale_calibrate(); // 采黑白阈值（车先放黑线上再放白布上） mod_encoder_reset(); // 编码器里程清零 mod_imu_calibrate(); // 陀螺仪零漂 state_init(); // state = S0, lap = 0 mod_oled_show("READY"); /* --- 等按键发车 --- */ while (BTN_START() == 0); /* --- 主循环 --- */ while (1) { task_control(); // 读 → 算 → 出 → 判，每 10ms 一次 // 这里加一个 10ms 延时（或放定时器中断里触发） } }
总结：整个程序的数据流向
8 路灰度 ADC
→ line_pos
编码器 TIM
→ distance, speed
MPU6050 I2C
→ heading
▼ 三个变量进入 task_control() ▼
方向环 PID
line_pos → turn_amount
→
速度环 PID ×2
(base±turn) vs actual → pwm_l, pwm_r
▼ 两路 PWM ▼
四路编码器驱动
→ 520 电机左轮 + 右轮
▲ 编码器反馈实际速度 → 速度环闭环
